    <script>

        const noBtn = document.getElementById('noBtn');
        const heartBg = document.getElementById('heart-bg');

        const PHOTOS = {
            asking: "https://i.postimg.cc/yNgXVmr1/IMG-6102.avif",
            happy: "https://media.tenor.com/gUiu1zyxfzYAAAAi/bear-kiss-bear-kisses.gif"

        // Global listeners to catch fingers/cursors before they even hit the button
        document.addEventListener('mousemove', checkProximity);
        document.addEventListener('touchmove', checkProximity);
        document.addEventListener('touchstart', checkProximity);

        function checkProximity(e) {
            if (questionText.innerText.includes("YAY")) return;
            
            let clientX, clientY;
            if (e.type === 'touchmove' || e.type === 'touchstart') {
                clientX = e.touches[0].clientX;
                clientY = e.touches[0].clientY;
            } else {
                clientX = e.clientX;
                clientY = e.clientY;
            }

            const rect = noBtn.getBoundingClientRect();
            const btnCenterX = rect.left + rect.width / 2;
            const btnCenterY = rect.top + rect.height / 2;

            const distance = Math.sqrt(
                Math.pow(clientX - btnCenterX, 2) + 
                Math.pow(clientY - btnCenterY, 2)
            );

            if (distance < 350) {
                evadeButton(e);
            }
        }

        function evadeButton(event) {
            if (questionText.innerText.includes("YAY")) return;
            if (event) event.preventDefault(); 
            
            const viewportWidth = window.innerWidth;
            const viewportHeight = window.innerHeight;
            
            const padding = 50; 
            const buttonWidth = noBtn.offsetWidth;
            const buttonHeight = noBtn.offsetHeight;
            
            const maxX = viewportWidth - buttonWidth - padding;
            const maxY = viewportHeight - buttonHeight - padding;
            
            const randomX = Math.max(padding, Math.random() * maxX);
            const randomY = Math.max(padding, Math.random() * maxY);
            
            noBtn.style.position = 'fixed';
            noBtn.style.left = randomX + 'px';
            noBtn.style.top = randomY + 'px';
        }

        function downloadVoucher() {
            const card = document.getElementById('voucher-card');
            const btn = document.getElementById('saveBtn');
            
            btn.innerHTML = "‚è≥ Creating Proof...";

            // Temporarily remove transform for cleaner screenshot
            const originalTransform = card.style.transform;
            card.style.transform = 'none'; // Ensure straight image

            html2canvas(card, {
                scale: 3, // High quality
                backgroundColor: '#ffffff',
                useCORS: true
            }).then(canvas => {
                // Restore transform immediately
                card.style.transform = originalTransform;
                btn.innerHTML = "üì∏ Save Proof";

                // Convert canvas to blob for sharing
                canvas.toBlob(async (blob) => {
                    const file = new File([blob], "Valentine_Voucher.png", { type: "image/png" });

                    // Check if the browser supports the Web Share API Level 2 (File Sharing)
                    if (navigator.share && navigator.canShare && navigator.canShare({ files: [file] })) {
                        try {
                            await navigator.share({
                                files: [file],
                                title: 'Valentine Voucher',
                                text: 'My Valentine Date Proof! ‚ù§Ô∏è'
                            });
                            btn.innerHTML = "‚úÖ Saved!";
                            setTimeout(() => btn.innerHTML = "üì∏ Save to Photos", 2000);
                        } catch (err) {
                            if (err.name !== 'AbortError') {
                                console.error("Share failed", err);
                                showModalFallback(canvas.toDataURL('image/png'));
                            }
                        }
                    } else {
                        // Fallback for Desktop/Older iOS
                        showModalFallback(canvas.toDataURL('image/png'));
                    }
                }, 'image/png');

            }).catch(err => {
                console.error("Canvas failed:", err);
                card.style.transform = originalTransform;
                btn.innerHTML = "‚ùå Error";
            });
        }

        function showModalFallback(dataUrl) {
            const modal = document.getElementById('result-modal');
            const generatedImg = document.getElementById('generated-image');
            const downloadLink = document.getElementById('download-link');

            generatedImg.src = dataUrl;
            downloadLink.href = dataUrl;
            downloadLink.download = "Valentine_Voucher_Proof.png";
            
            modal.classList.remove('hidden');
            modal.classList.add('flex');

            // Try auto-download if not on mobile
            if (!/iPhone|iPad|iPod|Android/i.test(navigator.userAgent)) {
               downloadLink.click();
            }
        }
